# 开始

    git clone https://github.com/tomasr/molokai.git
    cd molokai/colors && mkdir -p ~/.vim/colors/
    mv molokai.vim ~/.vim/colors/
    vim ~/.vimrc

添加如下：

    syntax enable
    set background=dark
    colorscheme molokai


# 命令行颜色

开始：
    vim ~/.bash_profile

添加如下：
    export CLICOLOR=1
    export PS1='\[\033[01;33m\]\u@\h\[\033[01;31m\] \W\$\[\033[00m\] '
